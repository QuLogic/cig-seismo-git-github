\begin{frame}
 \frametitle{Useful Commands}

 \begin{itemize}
  \item \texttt{\command{git blame}} --- See who made changes to files
        line-by-line
  \item \texttt{\command{git commit -{}-amend}} --- Fix a previous commit (do
not        do this to public commits)
  \item \texttt{\command{git rebase [-{}-interactive]}} --- Re-create commits
        on top of another commit (do not do this to public commits)
  \item \texttt{\command{git cherry-pick}} --- Copy commits from one branch to
        another
  \item \texttt{\command{gitk}} --- Viewer for git history (built-in)
  \item \texttt{\command{gitg}} --- Viewer for git history (separate install)
 \end{itemize}
\end{frame}
